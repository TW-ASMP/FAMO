=== Query 2: retrieve related compliance docs ===

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX temp: <http://example.com/tempdata/>
PREFIX comp: <http://ontology.eil.utoronto.ca/FAMO/compliance/>
#details (in particular the select variables) TBD based upon specification of results view from TH

select ?name ?compliance ?procedure where { 
 temp:compliance_form_dummy_asset rdf:type ?assetclass.
    ?workplan comp:appliesToAssetType ?assetclass;
        comp:name ?name;
   		comp:activityType ?activitytype;
   		comp:hasProcedure ?procedure.
    BIND (EXISTS {
        ?workplan comp:supports [a comp:ComplianceRelatedSection].    
        } AS ?compliance)
}
